local c00lgui = {}
c00lgui.Version = "4.1"
c00lgui.Author = "Grok"
c00lgui.Keybind = Enum.KeyCode.Insert

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local TweenService = game:GetService("TweenService")
local SoundService = game:GetService("SoundService")

-- Base GUI (SEM Matrix)
local function CreateBase()
    local ScreenGui = Instance.new("ScreenGui")
    ScreenGui.Name = "c00lgui_NoBG"
    ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
    ScreenGui.ResetOnSpawn = false
    ScreenGui.DisplayOrder = 999

    -- Fundo preto s√≥lido (sem chuva)
    local BG = Instance.new("Frame")
    BG.Name = "SolidBG"
    BG.Parent = ScreenGui
    BG.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    BG.Size = UDim2.new(1, 0, 1, 0)
    BG.ZIndex = 0

    return ScreenGui
end

-- Criar Janela com TABS
function c00lgui:Create(title, callback)
    local ScreenGui = CreateBase()
    local MainFrame = Instance.new("Frame")
    MainFrame.Name = "MainFrame"
    MainFrame.Parent = ScreenGui
    MainFrame.BackgroundColor3 = Color3.fromRGB(10, 10, 10)  -- Preto suave
    MainFrame.BorderSizePixel = 0
    MainFrame.Position = UDim2.new(0.02, 0, 0.1, 0)
    MainFrame.Size = UDim2.new(0, 380, 0, 550)
    MainFrame.Active = true
    MainFrame.Draggable = true
    MainFrame.Visible = false

    local Glow = Instance.new("UIStroke", MainFrame)
    Glow.Color = Color3.fromRGB(0, 255, 0)
    Glow.Thickness = 3
    Glow.Transparency = 0.4

    local Corner = Instance.new("UICorner", MainFrame)
    Corner.CornerRadius = UDim.new(0, 16)

    local Title = Instance.new("TextLabel")
    Title.Parent = MainFrame
    Title.BackgroundTransparency = 1
    Title.Size = UDim2.new(1, 0, 0, 50)
    Title.Font = Enum.Font.SourceSansBold
    Title.Text = "c00lgui | " .. title
    Title.TextColor3 = Color3.fromRGB(0, 255, 0)
    Title.TextScaled = true
    Title.TextStrokeTransparency = 0

    local CloseBtn = Instance.new("TextButton")
    CloseBtn.Parent = MainFrame
    CloseBtn.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
    CloseBtn.Position = UDim2.new(1, -55, 0, 8)
    CloseBtn.Size = UDim2.new(0, 45, 0, 35)
    CloseBtn.Font = Enum.Font.SourceSansBold
    CloseBtn.Text = "X"
    CloseBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
    CloseBtn.TextScaled = true
    local CloseCorner = Instance.new("UICorner", CloseBtn)
    CloseCorner.CornerRadius = UDim.new(0, 8)

    -- Tab Buttons
    local TabButtons = Instance.new("Frame")
    TabButtons.Parent = MainFrame
    TabButtons.BackgroundTransparency = 1
    TabButtons.Position = UDim2.new(0, 15, 0, 60)
    TabButtons.Size = UDim2.new(1, -30, 0, 40)

    local TabList = Instance.new("UIListLayout")
    TabList.Parent = TabButtons
    TabList.FillDirection = Enum.FillDirection.Horizontal
    TabList.Padding = UDim.new(0, 5)

    -- Content
    local ContentFrame = Instance.new("Frame")
    ContentFrame.Parent = MainFrame
    ContentFrame.BackgroundTransparency = 1
    ContentFrame.Position = UDim2.new(0, 15, 0, 105)
    ContentFrame.Size = UDim2.new(1, -30, 1, -120)

    local ui = {}
    local Tabs = {}
    local ActiveTab = nil

    function ui:Tab(name, callback)
        local TabBtn = Instance.new("TextButton")
        TabBtn.Parent = TabButtons
        TabBtn.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
        TabBtn.Size = UDim2.new(0, 100, 1, 0)
        TabBtn.Font = Enum.Font.SourceSansBold
        TabBtn.Text = name
        TabBtn.TextColor3 = Color3.fromRGB(180, 180, 180)
        TabBtn.TextScaled = true

        local TabCorner = Instance.new("UICorner", TabBtn)
        TabCorner.CornerRadius = UDim.new(0, 8)
        local TabStroke = Instance.new("UIStroke", TabBtn)
        TabStroke.Color = Color3.fromRGB(0, 255, 0)
        TabStroke.Thickness = 0

        local Scroll = Instance.new("ScrollingFrame")
        Scroll.Parent = ContentFrame
        Scroll.BackgroundTransparency = 1
        Scroll.Size = UDim2.new(1, 0, 1, 0)
        Scroll.CanvasSize = UDim2.new(0, 0, 0, 0)
        Scroll.ScrollBarThickness = 6
        Scroll.ScrollBarImageColor3 = Color3.fromRGB(0, 255, 0)
        Scroll.Visible = false

        local List = Instance.new("UIListLayout")
        List.Parent = Scroll
        List.Padding = UDim.new(0, 8)
        List:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            Scroll.CanvasSize = UDim2.new(0, 0, 0, List.AbsoluteContentSize.Y + 20)
        end)

        local tab = {}

        function tab:Toggle(name, default, callback)
            local state = default or false
            local Btn = Instance.new("TextButton")
            Btn.Parent = Scroll
            Btn.BackgroundColor3 = Color3.fromRGB(18, 18, 18)
            Btn.Size = UDim2.new(1, 0, 0, 45)
            Btn.Font = Enum.Font.SourceSansBold
            Btn.Text = name .." [OFF]"
            Btn.TextColor3 = Color3.fromRGB(255, 70, 70)
            Btn.TextScaled = true

            local Corner = Instance.new("UICorner", Btn)
            Corner.CornerRadius = UDim.new(0, 8)
            local Stroke = Instance.new("UIStroke", Btn)
            Stroke.Color = Color3.fromRGB(255, 70, 70)
            Stroke.Thickness = 2

            Btn.MouseButton1Click:Connect(function()
                state = not state
                Btn.Text = name .. (state and " [ON]" or " [OFF]")
                Btn.TextColor3 = state and Color3.fromRGB(0, 255, 0) or Color3.fromRGB(255, 70, 70)
                Stroke.Color = state and Color3.fromRGB(0, 255, 0) or Color3.fromRGB(255, 70, 70)
                pcall(callback, state)
                local s = Instance.new("Sound")
                s.SoundId = "rbxassetid://131961136"
                s.Volume = 0.3
                s.Parent = SoundService
                s:Play()
                s.Ended:Connect(function() s:Destroy() end)
            end)

            if default then task.spawn(function() Btn.MouseButton1Click:Fire() end) end
        end

        function tab:Button(name, callback)
            local Btn = Instance.new("TextButton")
            Btn.Parent = Scroll
            Btn.BackgroundColor3 = Color3.fromRGB(0, 140, 0)
            Btn.Size = UDim2.new(1, 0, 0, 40)
            Btn.Font = Enum.Font.SourceSansBold
            Btn.Text = name
            Btn.TextColor3 = Color3.fromRGB(255, 255, 255)
            Btn.TextScaled = true

            local Corner = Instance.new("UICorner", Btn)
            Corner.CornerRadius = UDim.new(0, 8)

            Btn.MouseButton1Click:Connect(function()
                pcall(callback)
                local s = Instance.new("Sound")
                s.SoundId = "rbxassetid://5357159749"
                s.Volume = 0.5
                s.Parent = SoundService
                s:Play()
                s.Ended:Connect(function() s:Destroy() end)
            end)
        end

        function tab:Slider(name, min, max, default, callback)
            local Frame = Instance.new("Frame")
            Frame.Parent = Scroll
            Frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
            Frame.Size = UDim2.new(1, 0, 0, 55)
            local Corner = Instance.new("UICorner", Frame)
            Corner.CornerRadius = UDim.new(0, 8)

            local Label = Instance.new("TextLabel")
            Label.Parent = Frame
            Label.BackgroundTransparency = 1
            Label.Size = UDim2.new(1, 0, 0.5, 0)
            Label.Font = Enum.Font.SourceSans
            Label.Text = name .. ": " .. default
            Label.TextColor3 = Color3.fromRGB(0, 255, 0)
            Label.TextScaled = true

            local Box = Instance.new("TextBox")
            Box.Parent = Frame
            Box.Position = UDim2.new(0.05, 0, 0.5, 0)
            Box.Size = UDim2.new(0.9, 0, 0.4, 0)
            Box.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
            Box.Text = tostring(default)
            Box.TextColor3 = Color3.fromRGB(0, 255, 0)
            Box.Font = Enum.Font.SourceSans

            Box.FocusLost:Connect(function()
                local val = tonumber(Box.Text) or default
                val = math.clamp(val, min, max)
                Box.Text = tostring(val)
                Label.Text = name .. ": " .. val
                pcall(callback, val)
            end)
        end

        TabBtn.MouseButton1Click:Connect(function()
            if ActiveTab then
                ActiveTab.Scroll.Visible = false
                ActiveTab.Btn.TextColor3 = Color3.fromRGB(180, 180, 180)
                ActiveTab.Btn.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
                TweenService:Create(ActiveTab.BtnStroke, TweenInfo.new(0.2), {Thickness = 0}):Play()
            end
            Scroll.Visible = true
            TabBtn.TextColor3 = Color3.fromRGB(0, 255, 0)
            TabBtn.BackgroundColor3 = Color3.fromRGB(10, 45, 10)
            TweenService:Create(TabStroke, TweenInfo.new(0.2), {Thickness = 2}):Play()
            ActiveTab = {Scroll = Scroll, Btn = TabBtn, BtnStroke = TabStroke}
        end)

        if #Tabs == 0 then
            task.spawn(function() TabBtn.MouseButton1Click:Fire() end)
        end

        table.insert(Tabs, {Scroll = Scroll, Btn = TabBtn})

        if callback then task.spawn(function() callback(tab) end) end

        return tab
    end

    CloseBtn.MouseButton1Click:Connect(function()
        MainFrame.Visible = not MainFrame.Visible
    end)

    UserInputService.InputBegan:Connect(function(input)
        if input.KeyCode == c00lgui.Keybind then
            MainFrame.Visible = not MainFrame.Visible
        end
    end)

    if callback then task.spawn(function() callback(ui) end) end

    return ui
end

print("c00lgui v4.1 (SEM FUNDO) carregado! Pressione INSERT.")
return c00lgui
